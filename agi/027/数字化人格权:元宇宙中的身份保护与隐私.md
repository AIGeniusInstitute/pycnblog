# 数字化人格权计算器

def calculate_dpr(id_score, dc_score, pp_score, pe_score):
    """
    计算用户数字化人格权指数

    Args:
        id_score: 身份识别权得分
        dc_score: 数据控制权得分
        pp_score: 隐私保护权得分
        pe_score: 财产权益权得分

    Returns:
        数字化人格权指数
    """
    alpha = 0.4
    beta = 0.3
    gamma = 0.2
    delta = 0.1
