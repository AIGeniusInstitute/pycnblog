### 5.2 源代码详细实现

以下是虚拟资产传承智能合约的源代码实现：

```solidity
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract VirtualAsset is ERC721, Ownable {
    using Strings for uint256;

    struct Asset {
        string name;
        string description;
        uint256 value;
        address owner;
    }

    mapping(uint256 => Asset) public assets;

    event AssetCreated(uint256 indexed id, string name, string description, uint256 value, address owner);
    event AssetTransferred(uint256 indexed id, address from, address to, uint256 value);

    constructor() ERC721("VirtualAsset", "VA") {}

    function createAsset(string memory name, string memory description, uint256 value) public onlyOwner returns (uint256) {
        uint256 id = _tokenIds.current;
        assets[id] = Asset(name, description, value, msg.sender);
        _mint(msg.sender, id);
        emit AssetCreated(id, name, description, value, msg.sender);
        return id;
    }

    function transferAsset(uint256 id, address to, uint256 value) public {
        require(ownerOf(id) == msg.sender, "Not the owner");
        _burn(id);
        assets[id].owner = to;
        assets[id].value = value;
        _mint(to, id);
        emit AssetTransferred(id, msg.sender, to, value);
    }
}
```

虚拟资产传承智能合约使用ERC721标准来实现NFT，使用Ownable合约来管理所有权。合约定义了两个函数：`createAsset`和`transferAsset`。`createAsset`函数用于创建新的数字资产，`transferAsset`函数用于转移数字资产的所有权。
