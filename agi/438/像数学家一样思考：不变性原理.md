```python
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                swapped = True
        if not swapped:
            break
    return arr

def test_bubble_sort():
    arr = [64, 25, 12, 22, 11]
    sorted_arr = bubble_sort(arr)
    assert sorted_arr == [11, 12, 22, 25, 64], "排序失败"
    print("测试通过")

test_bubble_sort()
```

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        # 不变性条件：每次循环结束后，至少有一个元素处于正确的位置
        swapped = False
