- 利用激光测距公式，计算太空战舰与目标之间的距离：

def solve_orbit(mass1, mass2, r0, v0):

    G = 6.67430e-11
    # 引力加速度
    a = G * mass2 / r0**2
    # 运动方程

    y = integrate.solve_ivp(dy_dt, tspan, y0, dense_output=True).y
    return y
