import cv2
import numpy as np

def detect_behavior(image):
    # 载入预训练的模型和权重
    model = cv2.face.EigenFaceRecognizer_create()
    model.read('model.yml')

    # 读取图像
    img = cv2.imread(image)

    # 调整图像大小
    img = cv2.resize(img, (160, 160))

    # 进行人脸检测
    face_cascade = cv2.CascadeClassifier('haarcascade_frontalface_default.xml')
    faces = face_cascade.detectMultiScale(img, 1.3, 5)

    # 遍历检测到的人脸
    for (x, y, w, h) in faces:
        # 提取人脸区域
        face_region = img[y:y+h, x:x+w]

        # 进行行为识别
        pred, conf = model.predict(face_region)

        # 输出结果
        print(f'Person detected: {pred}, Confidence: {conf}')

# 调用函数
detect_behavior('example.jpg')

import cv2
import numpy as np

def detect_behavior(image):
    # Load the pre-trained model and weights
    model = cv2.face.EigenFaceRecognizer_create()
    model.read('model.yml')

    # Read the image
    img = cv2.imread(image)

    # Resize the image
    img = cv2.resize(img, (160, 160))

    # Detect faces
    face_cascade = cv2.CascadeClassifier('haarcascade_frontalface_default.xml')
    faces = face_cascade.detectMultiScale(img, 1.3, 5)

    # Iterate through detected faces
    for (x, y, w, h) in faces:
        # Extract the face region
        face_region = img[y:y+h, x:x+w]

        # Perform behavior recognition
        pred, conf = model.predict(face_region)

        # Output the result
        print(f'Person detected: {pred}, Confidence: {conf}')

# Call the function
detect_behavior('example.jpg')
```

1. **Import Libraries**: Import necessary libraries, including OpenCV and NumPy.

2. **Behavior Recognition Function**: Define a function named `detect_behavior` that takes an image file path as input.

3. **Model Loading**: Load the pre-trained face recognition model and weights.

4. **Image Preprocessing**: Read the image, resize it, and perform face detection.

5. **Behavior Recognition**: Recognize the behavior of the detected face regions and output the results.
