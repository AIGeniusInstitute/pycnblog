pragma solidity ^0.8.0;

contract VirtualAssetInheritance {
    address public owner;
    mapping(address => uint256) public assets;

    event AssetTransferred(address recipient, uint256 amount);

    constructor() public {
        owner = msg.sender;
        assets[owner] = 100; // 初始化虚拟资产数量为100
    }

    function transferAsset(address recipient, uint256 amount) public {
        require(msg.sender == owner, "只有资产所有者可以调用此函数");
        require(assets[owner] >= amount, "虚拟资产数量不足");

        assets[owner] -= amount;
        assets[recipient] += amount;

        emit AssetTransferred(recipient, amount);
    }
}
```

```solidity
pragma solidity ^0.8.0;

contract VirtualAssetInheritance {
    address public owner;
    mapping(address => uint256) public assets;

    event AssetTransferred(address recipient, uint256 amount);

    constructor() public {
        owner = msg.sender;
        assets[owner] = 100; // 初始化虚拟资产数量为100
    }

    function transferAsset(address recipient, uint256 amount) public {
        require(msg.sender == owner, "只有资产所有者可以调用此函数");
        require(assets[owner] >= amount, "虚拟资产数量不足");

        assets[owner] -= amount;
        assets[recipient] += amount;

        emit AssetTransferred(recipient, amount);
    }
}
