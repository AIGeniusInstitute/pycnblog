### 4.2 智能合约的举例说明

以下是一个简单的智能合约示例，用于管理一个共享资金池：

```solidity
pragma solidity ^0.8.0;

contract FundPool {
    address owner;
    uint256 totalBalance;
    mapping(address => uint256) contributions;

    constructor() {
        owner = msg.sender;
        totalBalance = 0;
    }

    function contribute() public payable {
        contributions[msg.sender] += msg.value;
        totalBalance += msg.value;
    }

    function distributeFunds() public {
        require(msg.sender == owner, "Only the owner can distribute funds.");
        for (address contributor : contributors) {
            uint256 contribution = contributions[contributor];
            payable(contributor).transfer(contribution);
        }
        totalBalance = 0;
    }
}
```

在这个示例中，`contribute()` 函数用于向资金池贡献资金，`distributeFunds()` 函数用于将资金池中的资金分发给所有贡献者。
