以下是数字化遗产项目的源代码实现：

```python
# 导入必要的库
from blockchain import Blockchain
from cryptocurrency import encrypt, decrypt
from gan import Generator, Discriminator
import tensorflow as tf

# 初始化区块链
blockchain = Blockchain()

# 初始化加密库
crypto = encrypt()

# 初始化GAN模型
generator = Generator()
discriminator = Discriminator()

# 定义用户注册函数
def register_user(info):
    # 加密用户信息
    encrypted_info = crypto.encrypt(info)
    # 将用户信息添加到区块链
    blockchain.add_block(encrypted_info)

# 定义头像生成函数
def generate_avatar(photo):
    # 使用GAN生成头像
    avatar = generator.generate(photo)
    # 将头像保存到区块链
    blockchain.add_block(avatar)

# 定义技能和成就记录函数
def record_achievements(achievements):
    # 加密成就信息
    encrypted_achievements = crypto.encrypt(achievements)
    # 将成就信息添加到区块链
    blockchain.add_block(encrypted_achievements)

# 测试代码
# 注册用户
register_user("Alice")
# 生成头像
generate_avatar("Alice's photo")
# 记录技能和成就
record_achievements("Alice's achievements")

# 查看区块链数据
blockchain.print_chain()
```
