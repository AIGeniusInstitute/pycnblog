import numpy as np

def system_model(x, u):
    A = np.array([[0.9, 0.1], [-0.1, 0.9]])
    B = np.array([[1], [1]])
    x_next = A @ x + B @ u
    return x_next

def simulate_system(u0, A, B, T):
    x = np.zeros((T, 2))
    x[0] = u0
    for i in range(1, T):
        x[i] = system_model(x[i-1], u)
    return x

# 仿真测试
