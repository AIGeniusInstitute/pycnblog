import numpy as np

# 定义符号变量

# Kerr解的参数
a = m/c

# 构建Kerr解的表达式
kerr_metric = sp.Matrix([
    [-(1 - 2*m/r) + (a**2*c**2/(r**2)), a*c*(sp.cos(theta)), 0],

    [0, -a*c*sp.sin(theta), -(r**2 - a**2)]

# 计算度量的黎曼张量
ricci_tensor = kerr_metric.ricci_tensor()

# 输出Kerr解的黎曼张量表达式
