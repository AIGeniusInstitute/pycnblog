import torch
import brainpy as bp

class CognitiveModel(bp.Net):
    def __init__(self, *args, **kwargs):
        super(CognitiveModel, self).__init__(*args, **kwargs)
        self.fc1 = bp.layers.Linear(784, 128)
        self.fc2 = bp.layers.Linear(128, 10)

    def forward(self, x):
        x = self.fc1(x)
        x = bp.relu(x)
        x = self.fc2(x)
        return x

# 初始化模型和优化器
