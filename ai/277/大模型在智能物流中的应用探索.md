class Transformer(nn.Module):
    def __init__(self, d_model, nhead, dim_ff, dropout=0.1, batch_first=True):
        super(Transformer, self).__init__()
        self.encoder_layer = nn.TransformerEncoderLayer(d_model, nhead, dim_ff, dropout, batch_first)
        self.encoder = nn.TransformerEncoder(self.encoder_layer, num_layers=6)

    def forward(self, src):
        output = self.encoder(src)
        return output

def train(model, optimizer, criterion, data_loader, device):
    model.train()
    total_loss = 0
    for batch in data_loader:
        src, target = batch
        src, target = src.to(device), target.to(device)
        optimizer.zero_grad()
        output = model(src)
        loss = criterion(output, target)
        loss.backward()
        optimizer.step()
        total_loss += loss.item()
    return total_loss / len(data_loader)

def evaluate(model, data_loader, device):
    model.eval()
    total_loss = 0

            loss = criterion(output, target)
            total_loss += loss.item()
    return total_loss / len(data_loader)
