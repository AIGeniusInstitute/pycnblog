### 5.2 源代码详细实现

```python
import numpy as np
import pandas as pd

# 人才需求预测模型
def talent_demand_prediction(project_size, technical_difficulty, retention_rate):
    demand = project_size * technical_difficulty * retention_rate
    return demand

# 人才流失预测模型

    attrition_rate = quit_rate + retire_rate + transfer_rate

print("人才需求量：", demand)
print("人才流失率：", attrition_rate)
