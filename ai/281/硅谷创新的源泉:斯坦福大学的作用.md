import networkx as nx
import numpy as np
import scipy.sparse as sp

# 创建创新者集合和资源集合
V = ['L', 'S']
E = ['U', 'C', 'F']  # U: University, C: Capital, F: Facilities

# 创建创新者与资源之间的无向图
G = nx.Graph()
G.add_nodes_from(V + E)

# 添加连边
G.add_edge('L', 'U')
G.add_edge('L', 'C')
G.add_edge('S', 'U')
G.add_edge('S', 'F')

# 创建创新成功可能性函数
def f(V, M, E, p, r):
    return p * np.prod(r)

# 创建市场集合和环境集合
M = ['Search Engine']
E = ['Internet']

# 创建创新者与市场和环境的关系强度向量
r = np.array([0.6, 0.7, 0.8])

# 计算创新成功可能性
R = f(V, M, E, p=0.9, r=r)
print("创新成功可能性：", R)
