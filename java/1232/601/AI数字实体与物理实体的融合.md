H_1 = np.round(H, 2)
CO2_1 = np.round(CO2, 2)

# 数据存储
T_2 = pd.DataFrame(T_1, columns=['Temperature'])
H_2 = pd.DataFrame(H_1, columns=['Humidity'])
CO2_2 = pd.DataFrame(CO2_1, columns=['CO2'])
