$$

7 & 8 & 9
\end{bmatrix}

矩阵中的每个数字称为矩阵的元素。矩阵的行数和列数分别称为矩阵的阶数。

import numpy as np
from scipy.linalg import qr

# 利用 qr 函数进行 QR 分解
